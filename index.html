Example Request 
// Buy MTN Data Bundle
const response = await fetch('https://api.zelvapay.com/v1/data', {
  method: 'POST',
  headers: {
    'Authorization': `Bearer ${API_KEY}`,
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    network: 'MTN',
    phone: '08012345678',
    plan_id: 'mtn_1gb_30days',
    amount: 300
  })
});

const data = await response.json();
